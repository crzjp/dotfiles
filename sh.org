** History

#+begin_src sh
shopt -s histappend

HISTCONTROL=ignoreboth:erasedups
HISTFILESIZE=10000
HISTSIZE=$HISTFILESIZE
HISTFILE=$HOME/.bash_history
#+end_src

** Vterm config

#+begin_src sh
if [[ "$INSIDE_EMACS" = 'vterm' ]] \
       && [[ -n ${EMACS_VTERM_PATH} ]] \
       && [[ -f ${EMACS_VTERM_PATH}/etc/emacs-vterm-bash.sh ]]; then
	source ${EMACS_VTERM_PATH}/etc/emacs-vterm-bash.sh
fi
#+end_src

** Start session

#+begin_src sh

[ "$(tty)" = "/dev/tty1" ] && sx
